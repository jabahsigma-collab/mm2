local function ClearAllESP()
   for player, highlight in pairs(CreatedHighlights) do
       if highlight and highlight.Parent then highlight:Destroy() end
   end
   table.clear(CreatedHighlights)
   
   for player, billGui in pairs(CreatedNames) do
       if billGui and billGui.Parent then billGui:Destroy() end
   end
   table.clear(CreatedNames)

   for player, tracerLine in pairs(CreatedTracers) do
       if tracerLine then tracerLine:Remove() end
   end
   table.clear(CreatedTracers)
end

local function GetRoleColor(role, isName)
   if isName and ESP_Rainbow_Names then return Color3.fromHSV(tick() % 5 / 5, 1, 1) end
   if not isName and ESP_Rainbow_Highlights then return Color3.fromHSV(tick() % 5 / 5, 1, 1) end
   if role == "Murderer" then return Colors.Murderer
   elseif role == "Sheriff" then return Colors.Sheriff
   elseif role == "Innocent" then return Colors.Innocent
   end
   return Colors.Default
end

local function UpdatePlayerESP(player, role)
   if not player or player == LocalPlayer then return end
   local character = player.Character
   if not character then return end
   local head = character:FindFirstChild("Head")
   local hrp = character:FindFirstChild("HumanoidRootPart")
   if not head or not hrp then return end

   local shouldHighlight = false
   if role == "Murderer" and ESP_Highlight_Murderer then shouldHighlight = true
   elseif role == "Sheriff" and ESP_Highlight_Sheriff then shouldHighlight = true
   elseif role == "Innocent" and ESP_Highlight_Innocent then shouldHighlight = true
   end

   local shouldName = false
   if role == "Murderer" and ESP_Name_Murderer then shouldName = true
   elseif role == "Sheriff" and ESP_Name_Sheriff then shouldName = true
   elseif role == "Innocent" and ESP_Name_Innocent then shouldName = true
   end

   local shouldTracer = false
   if role == "Murderer" and ESP_Tracers_Murderer then shouldTracer = true
   elseif role == "Sheriff" and ESP_Tracers_Sheriff then shouldTracer = true
   elseif role == "Innocent" and ESP_Tracers_Innocent then shouldTracer = true
   end

   local highlight = character:FindFirstChild("CustomESPHighlight")
   if shouldHighlight then
       if not highlight then
           highlight = Instance.new("Highlight")
           highlight.Name = "CustomESPHighlight"
           highlight.OutlineColor = Color3.fromRGB(255, 255, 255)
           highlight.FillTransparency = 0.4
           highlight.OutlineTransparency = 0
           highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
           highlight.Adornee = character
           highlight.Parent = character
           CreatedHighlights[player] = highlight
       end
       highlight.FillColor = GetRoleColor(role, false)
   else
       if highlight then highlight:Destroy() CreatedHighlights[player] = nil end
   end

   local billGui = head:FindFirstChild("CustomESPNameGui")
   if shouldName then
       if not billGui then
           billGui = Instance.new("BillboardGui")
           billGui.Name = "CustomESPNameGui"
           billGui.Size = UDim2.new(0, 200, 0, 50)
           billGui.StudsOffset = Vector3.new(0, 2.5, 0)
           billGui.AlwaysOnTop = true
           billGui.Adornee = head
           billGui.Parent = head
           
           local textLabel = Instance.new("TextLabel")
           textLabel.Name = "NameLabel"
           textLabel.Size = UDim2.new(1, 0, 1, 0)
           textLabel.BackgroundTransparency = 1
           textLabel.Text = player.Name
           textLabel.Font = Enum.Font.GothamBold
           textLabel.TextSize = 13
           textLabel.TextStrokeTransparency = 0
           textLabel.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
           textLabel.Parent = billGui
           CreatedNames[player] = billGui
       end
       local label = billGui:FindFirstChild("NameLabel")
       if label then label.TextColor3 = GetRoleColor(role, true) end
   else
       if billGui then billGui:Destroy() CreatedNames[player] = nil end
   end

   local tracerLine = CreatedTracers[player]
   if shouldTracer then
       local currentCamera = workspace.CurrentCamera
       if currentCamera then
           local screenPos, onScreen = currentCamera:WorldToViewportPoint(hrp.Position)
           if onScreen then
               if not tracerLine then
                   tracerLine = Drawing.new("Line")
                   tracerLine.Thickness = 1.5
                   tracerLine.Transparency = 1
                   CreatedTracers[player] = tracerLine
               end
               tracerLine.Visible = true
               tracerLine.From = Vector2.new(currentCamera.ViewportSize.X / 2, currentCamera.ViewportSize.Y)
               tracerLine.To = Vector2.new(screenPos.X, screenPos.Y)
               tracerLine.Color = GetRoleColor(role, false)
           else
               if tracerLine then tracerLine.Visible = false end
           end
       end
   else
       if tracerLine then tracerLine:Remove() CreatedTracers[player] = nil end
   end
end

local function StartESPLoop()
   if ESP_Loop_Active then return end
   ESP_Loop_Active = true
   task.spawn(function()
       while ESP_Enabled do
           for playerName, role in pairs(CurrentPlayersRoles) do
               local player = Players:FindFirstChild(playerName)
               if player then UpdatePlayerESP(player, role) end
           end
           task.wait(0.03)
       end
       ESP_Loop_Active = false
   end)
end

-- Объявление функций для ивентов до их использования
local OnFadeReceived, OnRoundEndReceived

function OnFadeReceived(data)
   if type(data) ~= "table" then return end
   ClearAllESP()
   task.wait(0.1)
   table.clear(CurrentPlayersRoles)
   CurrentMurdererName = ""
   for playerName, info in pairs(data) do
       if type(info) == "table" and info.Role then
           CurrentPlayersRoles[playerName] = info.Role
           if info.Role == "Murderer" then CurrentMurdererName = playerName end
       end
   end
end

function OnRoundEndReceived()
   table.clear(CurrentPlayersRoles)
   CurrentMurdererName = ""
   ClearAllESP()
end

local function CheckGlobalESPState()
   if ESP_Highlight_Murderer or ESP_Highlight_Sheriff or ESP_Highlight_Innocent or 
      ESP_Name_Murderer or ESP_Name_Sheriff or ESP_Name_Innocent or
      ESP_Tracers_Murderer or ESP_Tracers_Sheriff or ESP_Tracers_Innocent or
      ESP_Rainbow_Names or ESP_Rainbow_Highlights then
       if not ESP_Enabled then
           ESP_Enabled = true
           local Remotes = ReplicatedStorage:WaitForChild("Remotes", 5)
           local Gameplay = Remotes and Remotes:WaitForChild("Gameplay", 5)
           if Gameplay then
               local FadeRemote = Gameplay:WaitForChild("Fade", 5)
               local RoundEndRemote = Gameplay:WaitForChild("RoundEndFade", 5)
               if not Connection_Fade and FadeRemote then Connection_Fade = FadeRemote.OnClientEvent:Connect(OnFadeReceived) end
               if not Connection_RoundEnd and RoundEndRemote then Connection_RoundEnd = RoundEndRemote.OnClientEvent:Connect(OnRoundEndReceived) end
           end
           StartESPLoop()
       end
   else
       ESP_Enabled = false
       ClearAllESP()
   end
end

-- Циклы физики персонажа
local function StartCharacterLoop()
   if Connection_CharacterTick then return end
   Connection_CharacterTick = RunService.Heartbeat:Connect(function()
       local character = LocalPlayer.Character
       if not character then return end
       local humanoid = character:FindFirstChildOfClass("Humanoid")
       if not humanoid then return end

       if Main_BHop_Enabled and humanoid.FloorMaterial == Enum.Material.Air then
           humanoid.WalkSpeed = Main_BHop_Speed
       elseif Main_WalkSpeed_Enabled then
           humanoid.WalkSpeed = Main_WalkSpeed_Value
       else
           if humanoid.WalkSpeed == Main_WalkSpeed_Value or humanoid.WalkSpeed == Main_BHop_Speed then humanoid.WalkSpeed = 16 end
       end

       if Main_JumpPower_Enabled then
           humanoid.UseJumpPower = true
           humanoid.JumpPower = Main_JumpPower_Value
       else
           if humanoid.JumpPower == Main_JumpPower_Value then humanoid.JumpPower = 50 end
       end
   end)
end

local function UpdateAtmosphereFog()
   if Main_Fog_Enabled then
       Lighting.FogStart = 0
       Lighting.FogEnd = Main_Fog_Radius
       Lighting.FogColor = Color3.fromRGB(Main_Fog_R, Main_Fog_G, Main_Fog_B)
   else
       Lighting.FogStart = OriginalFogStart
       Lighting.FogEnd = OriginalFogEnd
       Lighting.FogColor = OriginalFogColor
   end
end

local function GetTargetHitboxPart(character, hitboxCategory)
   if not character then return nil end
   local possibleNames = HitboxMapping[hitboxCategory]
   if possibleNames then
       for _, partName in ipairs(possibleNames) do
           local part = character:FindFirstChild(partName)
           if part and part:IsA("BasePart") then return part end
       end
   end
   return character:FindFirstChild("HumanoidRootPart") or character:FindFirstChild("Head")
end

local function FireAtMurderer()
   local Gun = LocalPlayer:FindFirstChild("Backpack"):FindFirstChild("Gun") or (LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Gun"))
   if not Gun then return end
   local ShootRemote = Gun:FindFirstChild("Shoot")
   if not ShootRemote or not ShootRemote:IsA("RemoteEvent") then return end
   if CurrentMurdererName == "" then return end
   local murdererPlayer = Players:FindFirstChild(CurrentMurdererName)
   if not murdererPlayer or not murdererPlayer.Character then return end
   
   local targetPart = GetTargetHitboxPart(murdererPlayer.Character, Selected_Hitbox)
   if not targetPart then return end
   
   if ShootAI_Enabled then task.wait(ShootAI_Delay) end
   
   local originPosition = (LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")) and LocalPlayer.Character.HumanoidRootPart.Position or Vector3.new(0,0,0)
   local targetPosition = targetPart.Position
   
   if WallCheck_Enabled then
       local raycastParams = RaycastParams.new()
       raycastParams.FilterType = Enum.RaycastFilterType.Exclude
       raycastParams.FilterDescendantsInstances = {LocalPlayer.Character}
       local direction = (targetPosition - originPosition)
       local raycastResult = workspace:Raycast(originPosition, direction, raycastParams)
       if raycastResult and not raycastResult.Instance:IsDescendantOf(murdererPlayer.Character) and raycastResult.Instance.CanCollide == true then
           return
       end
   end
   
   ShootRemote:FireServer(CFrame.new(originPosition, targetPosition), CFrame.new(targetPosition))
end

local function FindGunDropPart()
   for _, obj in ipairs(workspace:GetDescendants()) do
       if obj:IsA("BasePart") and obj.Name == "GunDrop" then return obj end
   end
   return nil
end

local function TeleportDroppedGunToUs()
   if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then return end
   local gunPart = FindGunDropPart()
   if gunPart then gunPart.CFrame = LocalPlayer.Character.HumanoidRootPart.CFrame end
end

local function StartGunAuraLoop()
   if GunAura_Loop_Active then return end
   GunAura_Loop_Active = true
   task.spawn(function()
       while GunAura_Enabled do
           if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
               local gunPart = FindGunDropPart()
               if gunPart then
                   local distance = (gunPart.Position - LocalPlayer.Character.HumanoidRootPart.Position).Magnitude
                   if distance <= GunAura_Radius then gunPart.CFrame = LocalPlayer.Character.HumanoidRootPart.CFrame end
               end
           end
           task.wait(0.2)
       end
       GunAura_Loop_Active = false
   end)
end

-- Цикл автоматического взятия оружия (AutoGetGun)
local function StartAutoGetGunLoop()
   if AutoGetGun_Loop_Active then return end
   AutoGetGun_Loop_Active = true
   task.spawn(function()
       while AutoGetGun_Enabled do
           TeleportDroppedGunToUs()
           task.wait(0.1)
       end
       AutoGetGun_Loop_Active = false
   end)
end

local function SetupAdvancedResizing(targetButton)
   local InResizeMode = false
   local ResizeBorderSize = 7

   local Borders = {
       R = Instance.new("TextButton", targetButton),
       B = Instance.new("TextButton", targetButton),
       L = Instance.new("TextButton", targetButton),
       T = Instance.new("TextButton", targetButton),
       BR = Instance.new("TextButton", targetButton),
       BL = Instance.new("TextButton", targetButton),
       TR = Instance.new("TextButton", targetButton),
       TL = Instance.new("TextButton", targetButton)
   }

   for name, b in pairs(Borders) do
       b.BackgroundTransparency = 1
       b.Text = ""
       b.Visible = false
       b.ZIndex = 10
   end

   local function UpdateBorderPositions()
       Borders.R.Size = UDim2.new(0, ResizeBorderSize, 1, -ResizeBorderSize*2)
       Borders.R.Position = UDim2.new(1, -ResizeBorderSize, 0, ResizeBorderSize)
       
       Borders.B.Size = UDim2.new(1, -ResizeBorderSize*2, 0, ResizeBorderSize)
       Borders.B.Position = UDim2.new(0, ResizeBorderSize, 1, -ResizeBorderSize)
       
       Borders.L.Size = UDim2.new(0, ResizeBorderSize, 1, -ResizeBorderSize*2)
       Borders.L.Position = UDim2.new(0, 0, 0, ResizeBorderSize)
       
       Borders.T.Size = UDim2.new(1, -ResizeBorderSize*2, 0, ResizeBorderSize)
       Borders.T.Position = UDim2.new(0, ResizeBorderSize, 0, 0)
       
       Borders.BR.Size = UDim2.new(0, ResizeBorderSize, 0, ResizeBorderSize)
       Borders.BR.Position = UDim2.new(1, -ResizeBorderSize, 1, -ResizeBorderSize)
       
       Borders.BL.Size = UDim2.new(0, ResizeBorderSize, 0, ResizeBorderSize)
       Borders.BL.Position = UDim2.new(0, 0, 1, -ResizeBorderSize)
       
       Borders.TR.Size = UDim2.new(0, ResizeBorderSize, 0, ResizeBorderSize)
       Borders.TR.Position = UDim2.new(1, -ResizeBorderSize, 0, 0)
       
       Borders.TL.Size = UDim2.new(0, ResizeBorderSize, 0, ResizeBorderSize)
       Borders.TL.Position = UDim2.new(0, 0, 0, 0)
   end
   UpdateBorderPositions()

   targetButton:GetPropertyChangedSignal("Size"):Connect(UpdateBorderPositions)

   local function ToggleResizeMode()
       InResizeMode = not InResizeMode
       for _, b in pairs(Borders) do b.Visible = InResizeMode end
       targetButton.UIStroke.Color = InResizeMode and Color3.fromRGB(255, 165, 0) or (targetButton.Name == "ShootClickButton" and Color3.fromRGB(0, 120, 255) or Color3.fromRGB(0, 180, 120))
   end

   targetButton.MouseButton2Click:Connect(ToggleResizeMode)
   
   local touchStart = 0
   targetButton.InputBegan:Connect(function(input)
       if input.UserInputType == Enum.UserInputType.Touch then
           touchStart = tick()
       end
   end)
   targetButton.InputEnded:Connect(function(input)
       if input.UserInputType == Enum.UserInputType.Touch and touchStart > 0 then
           if tick() - touchStart > 1.2 then ToggleResizeMode() end
           touchStart = 0
       end
   end)

   for mode, btn in pairs(Borders) do
       local Dragging = false
       local StartPos, StartSize, MouseStart

       btn.InputBegan:Connect(function(input)
           if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
               Dragging = true
               StartPos = targetButton.Position
               StartSize = targetButton.Size
               MouseStart = input.Position
               input.Changed:Connect(function()
                   if input.UserInputState == Enum.UserInputState.End then Dragging = false end
               end)
           end
       end)

       UserInputService.InputChanged:Connect(function(input)
           if Dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
               local delta = input.Position - MouseStart
               local newW = StartSize.X.Offset
               local newH = StartSize.Y.Offset
               local newX = StartPos.X.Offset
               local newY = StartPos.Y.Offset

               if string.find(mode, "R") then newW = math.max(40, StartSize.X.Offset + delta.X) end
               if string.find(mode, "B") then newH = math.max(30, StartSize.Y.Offset + delta.Y) end
               
               if string.find(mode, "L") then 
                   local change = delta.X
                   if StartSize.X.Offset - change > 40 then
                       newW = StartSize.X.Offset - change
                       newX = StartPos.X.Offset + change
                   end
               end
               
               if string.find(mode, "T") then 
                   local change = delta.Y
                   if StartSize.Y.Offset - change > 30 then
                       newH = StartSize.Y.Offset - change
                       newY = StartPos.Y.Offset + change
                   end
               end

               targetButton.Size = UDim2.new(StartSize.X.Scale, newW, StartSize.Y.Scale, newH)
               targetButton.Position = UDim2.new(StartPos.X.Scale, newX, StartPos.Y.Scale, newY)
           end
       end)
   end
end

local Dragging, DragInput, DragStart, StartPosition
local function UpdateDrag(input)
   local delta = input.Position - DragStart
   MainFrame.Position = UDim2.new(StartPosition.X.Scale, StartPosition.X.Offset + delta.X, StartPosition.Y.Scale, StartPosition.Y.Offset + delta.Y)
end

MainFrame.InputBegan:Connect(function(input)
   if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
       Dragging = true
       DragStart = input.Position
       StartPosition = MainFrame.Position
       input.Changed:Connect(function()
           if input.UserInputState == Enum.UserInputState.End then Dragging = false end
       end)
   end
end)
MainFrame.InputChanged:Connect(function(input)
   if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then DragInput = input end
end)
UserInputService.InputChanged:Connect(function(input)
   if input == DragInput and Dragging then UpdateDrag(input) end
end)

local function ToggleMinimize()
   if IsMinimized then
       MainFrame.ClipsDescendants = true
       TweenService:Create(SidebarFrame, TweenInfo.new(0.2), {BackgroundTransparency = 0}):Play()
       TweenService:Create(SidebarPatch, TweenInfo.new(0.2), {BackgroundTransparency = 0}):Play()
       TweenService:Create(PagesContainer, TweenInfo.new(0.2), {BackgroundTransparency = 1}):Play()

       TweenService:Create(CloseButton, TweenInfo.new(0.2), {TextTransparency = 0}):Play()
       TweenService:Create(MinimizeButton, TweenInfo.new(0.2), {TextTransparency = 0}):Play()
       TweenService:Create(MainStroke, TweenInfo.new(0.3), {Thickness = 1.5}):Play()
       
       local sizeTween = TweenService:Create(MainFrame, TweenInfo.new(0.4, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {Size = OriginalSize})
       sizeTween:Play()
       sizeTween.Completed:Connect(function() if not IsMinimized then OpenIndicator.Visible = false end end)

       local sizeTween = TweenService:Create(MainFrame, TweenInfo.new(0.4, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {Size = UDim2.new(0, 550, 0, 0)})
       sizeTween:Play()

local function CreatePage(pageName)
   local PageScrollingFrame = Instance.new("ScrollingFrame", PagesContainer)
   PageScrollingFrame.Name = pageName .. "Page"
   PageScrollingFrame.Size = UDim2.new(1, 0, 1, 0)
   PageScrollingFrame.BackgroundTransparency = 1
   PageScrollingFrame.Visible = false
   PageScrollingFrame.CanvasSize = UDim2.new(0, 0, 0, 0)
   PageScrollingFrame.ScrollBarThickness = 3
   PageScrollingFrame.ScrollBarImageColor3 = Color3.fromRGB(60, 60, 65)
   
   local PagePadding = Instance.new("UIPadding", PageScrollingFrame)
   PagePadding.PaddingLeft = UDim.new(0, 15)
   PagePadding.PaddingTop = UDim.new(0, 15)
   PagePadding.PaddingRight = UDim.new(0, 15)
   
   local PageListLayout = Instance.new("UIListLayout", PageScrollingFrame)
   PageListLayout.SortOrder = Enum.SortOrder.LayoutOrder
   PageListLayout.Padding = UDim.new(0, 10)
   
   PageListLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
       PageScrollingFrame.CanvasSize = UDim2.new(0, 0, 0, PageListLayout.AbsoluteContentSize.Y + 30)
   end)
   
   local TabButton = Instance.new("TextButton", TabButtonsContainer)
   TabButton.Size = UDim2.new(1, -20, 0, 35)
   TabButton.Position = UDim2.new(0, 10, 0, 0)
   TabButton.BackgroundColor3 = Color3.fromRGB(40, 40, 45)
   TabButton.BackgroundTransparency = 1
   TabButton.Text = "  " .. pageName
   TabButton.TextColor3 = Color3.fromRGB(140, 140, 145)
   TabButton.Font = Enum.Font.GothamSemibold
   TabButton.TextSize = 12
   TabButton.TextXAlignment = Enum.TextXAlignment.Left
   
   Instance.new("UICorner", TabButton).CornerRadius = UDim.new(0, 6)
   local btnPadding = Instance.new("UIPadding", TabButton)
   btnPadding.PaddingLeft = UDim.new(0, 10)

   Pages[pageName] = PageScrollingFrame
   TabButtons[pageName] = TabButton
   
   if not FirstPage then FirstPage = pageName end
   
   TabButton.MouseButton1Click:Connect(function()
       for name, frame in pairs(Pages) do frame.Visible = (name == pageName) end
       for name, btn in pairs(TabButtons) do
           if name == pageName then
               btn.BackgroundTransparency = 0.4
               btn.TextColor3 = Color3.fromRGB(255, 255, 255)
           else
               btn.BackgroundTransparency = 1
               btn.TextColor3 = Color3.fromRGB(140, 140, 145)
           end
       end
   end)
   
   return PageScrollingFrame
end

local function AddSlider(pageFrame, text, min, max, defaultValue, valueSuffix, callback)
   local SliderFrame = Instance.new("Frame", pageFrame)
   SliderFrame.Size = UDim2.new(1, 0, 0, 50)
   SliderFrame.BackgroundColor3 = Color3.fromRGB(20, 20, 24)
   
   Instance.new("UICorner", SliderFrame).CornerRadius = UDim.new(0, 7)
   local stroke = Instance.new("UIStroke", SliderFrame)
   stroke.Color = Color3.fromRGB(35, 35, 40)
   
   local Label = Instance.new("TextLabel", SliderFrame)
   Label.Size = UDim2.new(0.7, 0, 0, 25)
   Label.Position = UDim2.new(0, 12, 0, 2)
   Label.BackgroundTransparency = 1
   Label.Text = text
   Label.TextColor3 = Color3.fromRGB(210, 210, 215)
   Label.Font = Enum.Font.GothamSemibold
   Label.TextSize = 13
   Label.TextXAlignment = Enum.TextXAlignment.Left
   
   local ValueLabel = Instance.new("TextLabel", SliderFrame)
   ValueLabel.Size = UDim2.new(0.25, 0, 0, 25)
   ValueLabel.Position = UDim2.new(0.7, 12, 0, 2)
   ValueLabel.BackgroundTransparency = 1
   ValueLabel.Text = tostring(defaultValue) .. " " .. valueSuffix
   ValueLabel.TextColor3 = Color3.fromRGB(0, 180, 255)
   ValueLabel.Font = Enum.Font.GothamBold
   ValueLabel.TextSize = 12
   ValueLabel.TextXAlignment = Enum.TextXAlignment.Right
   
   local SliderBarBackground = Instance.new("TextButton", SliderFrame)
   SliderBarBackground.Size = UDim2.new(1, -24, 0, 6)
   SliderBarBackground.Position = UDim2.new(0, 12, 0, 34)
   SliderBarBackground.BackgroundColor3 = Color3.fromRGB(40, 40, 45)
   SliderBarBackground.Text = ""
   Instance.new("UICorner", SliderBarBackground).CornerRadius = UDim.new(1, 0)
   
   local SliderBarFill = Instance.new("Frame", SliderBarBackground)
   local initPercent = math.clamp((defaultValue - min) / (max - min), 0, 1)
   SliderBarFill.Size = UDim2.new(initPercent, 0, 1, 0)
   SliderBarFill.BackgroundColor3 = Color3.fromRGB(0, 180, 255)
   Instance.new("UICorner", SliderBarFill).CornerRadius = UDim.new(1, 0)
   
   local function UpdateSliderInput(input)
       local xOffset = math.clamp(input.Position.X - SliderBarBackground.AbsolutePosition.X, 0, SliderBarBackground.AbsoluteSize.X)
       local percent = xOffset / SliderBarBackground.AbsoluteSize.X
       local rawValue = min + (percent * (max - min))
       
       local finalValue = math.floor(rawValue * 10 + 0.5) / 10
       if max - min > 10 then finalValue = math.floor(rawValue + 0.5) end
       
       SliderBarFill.Size = UDim2.new(percent, 0, 1, 0)
       ValueLabel.Text = tostring(finalValue) .. " " .. valueSuffix
       task.spawn(callback, finalValue)
   end
   
   local IsSliding = false
   SliderBarBackground.InputBegan:Connect(function(input)
       if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
           IsSliding = true
           UpdateSliderInput(input)
       end
   end)
   UserInputService.InputChanged:Connect(function(input)
       if IsSliding and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
           UpdateSliderInput(input)
       end
   end)
   UserInputService.InputEnded:Connect(function(input)
       if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then IsSliding = false end
   end)
end
